1. Pair

  • Pair dùng để ghép 2 kiểu dữ liệu (có thể khác nhau) lại thành một.

  • Thuộc thư viện #include <utility>

Sử dụng

    //Khai báo 1 Pair
    pair<string, int> my_pair;
    //Khai báo và Gán Giá trị
    pair<string, double> mine2("ToiLaPair", 1.23);

    // Gán giá trị
    my_pair.first = "toi";
    my_pair.second = 28;
    //Or
    my_pair = {"toi", 28};
    //Or
    my_pair = make_pair("toi", 28);

    //In ra màn hình
    cout << my_pair.first << ' ' << my_pair.second;

Note: Lúc này Pair giống như 1 kiểu dữ liệu, ta có thể sử dụng như 1 biến ta hay dùng.

  • Ví dụ
    //Swap
    pair <string, int> a, b;
    a = {"a", 1}; b = {"b", 2};
    swap(a, b);
    cout << a.first << ' ' << a.second; // Output: b 2

    // So sánh 2 Pair
    //Riêng từng thành phần
    if(a.first!=b.first) 
        cout << "string 2 biến khác nhau";
    else cout << "string giống nhau";
    //chung
    if(a==b) cout << "a giống b";

LƯU Ý:

Khi so sánh 2 pair, chương trình sẽ tự động so sánh giá trị first sau đó so sánh giá trị second.

Pair lồng nhau

    pair<int, pair<char, int> > p1;
    p1.first = 10;
    p1.second.first = 'A';
    p1.second.second = 20;

2. Vector & Iterator

a. Vector trong C++

  • Định nghĩa: Vector là một container động, giống mảng nhưng có thể tự động thay đổi kích thước.
  • Khai báo:
    vector<int> vec;
  • Thao tác cơ bản:
    • Thêm phần tử: vec.push_back(value);
    • Xóa phần tử: vec.erase(vec.begin() + index);
    • Kích thước vector: vec.size();
    • Truy cập phần tử: vec[index];

b. Iterator

  • Iterator là một đối tượng trỏ tới các phần tử trong container (Vector, List, Set,...).
  • Định nghĩa: Có 4 iterator chính trong vector :
  • Cách khai báo:
    vector<int> v = {28, 100, 30, 20, 15};
    vector<int>::iterator it1 = v.begin();
    cout << "v[0] = " << *it1 << '\n';
    vector<int>::reverse_iterator it2 = v.rbegin();
    cout << "v[4] = " << *it2 << '\n';
    //OR
    auto it = v.begin();
  • Các thao tác cơ bản:
    • Iterator hỗ trợ các toán tử ++, -- hoặc toán tử toán học để bạn có thể di chuyển iterator qua lại các phần tử trong vector.
    vector<int> v = {1, 2, 3, 4, 5};
vector<int>::iterator it = v.begin();
cout << "v[0] = " << *it << '\n';//1
++it;
cout << "v[1] = " << *it << '\n';//2
it += 2;
cout << "v[3] = " << *it << '\n';//4
--it;
cout << "v[2] = " << *it << '\n';//3

Note: Hàm sort cho Vector

//Mảng
sort(arr, arr + n);
//Vector
sort(vec.begin(), vec.end());
//Mặc định là sắp xếp tăng dần.
//Nếu muốn Sort giảm dần
sort(vec.begin(), vec.end(), greater<int>());
//OR
sort(vec.rbegin(), vec.rend());

3. lower_bound và upper_bound

  • Yêu cầu: Mảng/Vector phải được sắp xếp tăng dần trước.

  • lower_bound: Tìm vị trí đầu tiên trong mảng (đã sắp xếp) mà giá trị >= x.

     auto it = lower_bound(vec.begin(), vec.end(), x);
  • upper_bound: Tìm vị trí đầu tiên mà giá trị > x.
     auto it = upper_bound(vec.begin(), vec.end(), x);
  • Ví dụ minh họa:
     vector<int> vec = {1, 3, 5, 7, 9};
     int x = 5;
     auto it1 = lower_bound(vec.begin(), vec.end(), x); // it1 trỏ tới 5
     auto it2 = upper_bound(vec.begin(), vec.end(), x); // it2 trỏ tới 7

4. map

  • Lưu trữ cặp khóa-giá trị.
  • Khóa duy nhất và tự động sắp xếp (mặc định theo thứ tự tăng).

Cách sử dụng:

  • Khởi tạo:
    map<int, string> mp;
  • Thêm phần tử:
mp[1] = "One";
mp.insert({2, "Two"});
  • Truy cập:
    cout << mp[1]; //One
  • Duyệt:
  for (auto& p : m) 
    cout << p.first << " : " << p.second << endl;

So sánh Map và Unordered_map

map unordered_map
Tốc độ truy vấn O(logN) O(1) trung bình, O(N) tệ nhất
Thứ tự phần tử Có thứ tự (theo key) Không có thứ tự
Dùng khi nào? Khi cần sắp xếp key Khi chỉ cần truy vấn nhanh

5. set

  • Lưu trữ các phần tử duy nhất.
  • Các phần tử trong set tự động được sắp xếp theo thứ tự tăng (mặc định).
  • Các hàm như insert(), erase(), find() có độ phức tạp O(logN)

Cách sử dụng:

  • Khởi tạo:
set<int> s;
  • Thêm phần tử:
s.insert(10);
s.insert(20);
s.insert(30);
  • Truy cập:
auto it = s.find(20);
if (it != s.end()) 
    cout << *it << endl; // 20
  • Duyệt:
for (auto& elem : s) 
    cout << elem << endl;
  • Xóa phần tử:
s.erase(20);
  • Kiểm tra phần tử:
if (s.count(10) > 0) 
    cout << "10 tồn tại!" << endl;

Bài 1: Pair, Vector, Map

Cho một số nguyên dương n và một danh sách gồm n xâu.

Output:

  • In ra mỗi xâu trong danh sách cùng với tần suất xuất hiện của nó. Nếu hai xâu có cùng tần suất, in ra xâu nào đến trước trong bảng chữ cái trước.

Input:

5
apple
banana
apple
cherry
banana

Output:

apple 2
banana 2
cherry 1

Gợi ý giải:

#include <bits/stdc++.h>
using namespace std;
bool cmp(pair<string, int> a, pair<string, int> b) {
    if (a.second == b.second)
        return a.first < b.first;  // Nếu tần suất giống nhau, sắp xếp theo bảng chữ cái
    return a.second > b.second;  // Nếu không, sắp xếp theo tần suất giảm dần
}

int main() {
    int n; cin >> n;
    
    map<string, int> freqMap;  // map lưu tần suất của các xâu
    
    // Nhập các xâu và cập nhật tần suất
    for (int i = 0; i < n; i++) {
        string s; cin >> s;
        freqMap[s]++;
    }
    
    // Chuyển map thành vector các cặp (xâu, tần suất)
    vector<pair<string, int>> freqVec(freqMap.begin(), freqMap.end());
    
    // Sắp xếp vector theo tần suất giảm dần, nếu tần suất giống nhau thì theo thứ tự bảng chữ cái
    sort(freqVec.begin(), freqVec.end(), cmp);
    
    // In ra kết quả
    for (auto it : freqVec) {
        cout << it.first << " " << it.second << endl;
    }
    
    return 0;
}

Bài 2: Set

Nhập n, m. Nhập n phần tử riêng biệt của tập A. Nhập m phần tử riêng biệt của tập B. Tìm Giao của 2 tập hợp.

#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    set<int> a, b;
    for (int i = 0; i < n; i++) {
        int x; cin >> x;
        a.insert(x);
    }
    for (int i = 0; i < m; i++) {
        int x; cin >> x;
        b.insert(x);
    }
    
    vector<int> result;
    for (int x : a) {
        if (b.count(x)) result.push_back(x);
    }
    
    for (int x : result) cout << x << " ";
    return 0;
}

Bài 3: lower_bound và upper_bound

Cho một mảng số nguyên đã sắp xếp tăng dần gồm N phần tử và một số nguyên X. Đếm số cặp (i, j) sao cho arr[i] + arr[j] < X với i < j.

Input:
arr = [1, 2, 3, 5, 7]
X = 6

Output:
4

#include <bits/stdc++.h>
using namespace std;
int n, X;
int sol(vector<int>& arr) {
    int count = 0;  
    for (int i = 0; i < n - 1; i++) {
        auto it = upper_bound(arr.begin() + i + 1, arr.end(), X - arr[i] - 1);
        count += (it - (arr.begin() + i + 1));
    }
    return count;
}
int main() {
    //Input
    cin >> n;
    vector<int> arr(n);
    for(int i=0; i<n; i++) cin >> arr[i];
    cin >> X;
    cout << sol(arr);
    return 0;
}
